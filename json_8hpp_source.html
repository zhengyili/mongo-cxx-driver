71  void visit_value(const types::b_binary& value) {
72 
73  std::size_t b64_len;
74 
75  b64_len = (value.size / 3 + 1) * 4 + 1;
76  std::unique_ptr<char> b64(reinterpret_cast<char*>(operator new(b64_len)));
77  b64::ntop(value.bytes, value.size, b64.get(), b64_len);
78 
79  out << "{" << std::endl;
80  pad(1);
81  out << "\"$type\" : " << to_string(value.sub_type) << "," << std::endl;
82  pad(1);
83  out << "\"$binary\" : " << b64.get() << "," << std::endl;
84  pad();
85  out << "}";
86  }

198  void visit_children(const document::view& view) {
199  bool first = true;
200  for (auto&& x : view) {
201  if (!first) {
202  out << ", " << std::endl;
203  }
204  first = false;
205  visit_key(x.key());
206  switch (static_cast<int>(x.type())) {
207 #define BSONCXX_ENUM(name, val) \
208  case val: \
209  visit_value(x.get_##name()); \
210  break;
211 #include <bsoncxx/enums/type.hpp>
212 #undef BSONCXX_ENUM
213  }
214  }
215  out << std::endl;
216  stack.pop_back();
217  }
