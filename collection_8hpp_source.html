601 template<typename document_view_iterator_type>
602 inline bsoncxx::stdx::optional<result::insert_many> collection::insert_many(
603  document_view_iterator_type begin,
604  document_view_iterator_type end,
605  const options::insert& options
606 ) {
607  class bulk_write writes(false);
608 
609  std::map<std::size_t, bsoncxx::document::element> inserted_ids{};
610  std::size_t index = 0;
611  std::for_each(begin, end, [&](const bsoncxx::document::view& current){
612  // TODO: put this somewhere else not in header scope (bsoncxx::builder)
613  if ( !current["_id"]) {
614  bsoncxx::builder::stream::document new_document;
615  new_document << "_id" << bsoncxx::oid(bsoncxx::oid::init_tag)
616  << bsoncxx::builder::stream::concatenate{current};
617 
618  writes.append(model::insert_one(new_document.view()));
619 
620  inserted_ids.emplace(index++, new_document.view()["_id"]);
621  } else {
622  writes.append(model::insert_one(current));
623 
624  inserted_ids.emplace(index++, current["_id"]);
625  }
626 
627  });
628 
629  if (options.write_concern())
630  writes.write_concern(*options.write_concern());
631  result::bulk_write res(std::move(bulk_write(writes).value()));
632  bsoncxx::stdx::optional<result::insert_many> result(result::insert_many(std::move(res), std::move(inserted_ids)));
633  return result;
634 }
